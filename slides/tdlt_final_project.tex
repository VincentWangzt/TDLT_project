\section{Problem Setup}

\begin{frame}{Why Predict Loss Curves?}
  Loss-curve prediction tries to answer a practical planning question before paying for a full training run.
  \begin{itemize}
    \item If we know a loss curve under one schedule, can we predict another schedule?
    \item If schedule choice affects extrapolation, scaling-law fits are not only about model/data size.
    \item Accurate prediction can reduce expensive trial-and-error in pretraining schedule design.
  \end{itemize}
  \smallgap
  The key challenge is that a learning-rate schedule affects the entire optimization history, not only the current step.
\end{frame}

\begin{frame}{Learning-Rate Schedule As Input}
  For total training length $T$, define a nonnegative learning-rate vector
  \[
    \bm{\eta}=(\eta_1,\eta_2,\ldots,\eta_T)^\top \in \mathbb{R}_{\ge 0}^T .
  \]
  The pretraining loss at step $t$ is treated as a functional of the schedule prefix:
  \[
    L_t = L_{\Theta}(\eta_{\le t}),
    \qquad
    \eta_{\le t}=(\eta_1,\eta_2,\ldots,\eta_t).
  \]
  We fit $\Theta$ from observed loss curves and use it to predict unseen schedules.
\end{frame}

\begin{frame}{Course Protocol}
  The strict main protocol is:
  \[
    \text{fit on cosine LRS} \quad \longrightarrow \quad \text{predict WSD LRS}.
  \]
  \begin{itemize}
    \item Fit target: EMA-smoothed loss with span 201.
    \item Evaluation region: post-warmup, implemented as \code{step >= 1000}.
    \item Evaluation grid: every 20 recorded steps plus the final point.
    \item Auxiliary schedule: 8-1-1, used only for additional evidence.
  \end{itemize}
\end{frame}

\begin{frame}{Main Evaluation Metric}
  The headline number is WSD full-region RMSE on the EMA target:
  \[
    \mathrm{RMSE}
      =
      \left(\frac{1}{|I|}
      \sum_{t\in I}
      \left(\widehat L_t - L^{\mathrm{EMA}}_t\right)^2
      \right)^{1/2}.
  \]
  We also report:
  \begin{itemize}
    \item MAE and MAPE for average absolute error.
    \item $R^2$ for explained variance.
    \item FinalE and WorstE for endpoint and worst-case behavior.
    \item Tail20/decay/final1000 regional errors as diagnostic views.
  \end{itemize}
\end{frame}

\begin{frame}{Central Difficulty}
  The data regime is extremely small.
  \begin{itemize}
    \item Only one cosine curve is available for the main fit.
    \item WSD is the main test curve, but it is tempting to use it during method development.
    \item 8-1-1 is useful, but it is not part of the official main protocol.
  \end{itemize}
  \smallgap
  This makes overparameterized formulas fragile: a lower cosine fitting loss can easily hurt schedule transfer.
\end{frame}

\section{Baselines}

\begin{frame}{Baseline 1: Momentum Law (MTL)}
  The Tissue et al. learning-rate annealing law is implemented as
  \[
    \widehat L(t)=L_0 + A\,T_t^{-\alpha} - C\,S_2(t),
    \qquad
    T_t = \epsilon + \sum_{i\le t}\eta_i .
  \]
  The annealing memory is
  \[
    m_i=\lambda m_{i-1}+(\eta_{i-1}-\eta_i),
    \qquad
    S_2(t)=\sum_{i\le t}m_i .
  \]
  We grid-search $\lambda$ and fit $(L_0,A,C,\alpha)$ by log-Huber loss.
\end{frame}

\begin{frame}{MTL Strengths And Weaknesses}
  \textbf{Strengths}
  \begin{itemize}
    \item Few parameters and stable fitting.
    \item Strong inductive bias for schedule memory.
    \item Best clean baseline on WSD EMA full RMSE.
  \end{itemize}
  \textbf{Weaknesses}
  \begin{itemize}
    \item A single exponential memory scale can miss multi-timescale schedule effects.
    \item Grid choice for $\lambda$ strongly affects transfer.
  \end{itemize}
\end{frame}

\begin{frame}{Baseline 2: Multi-Power Law (MPL)}
  MPL models schedule response with a nonlinear accumulated LR-drop term:
  \[
    \widehat L(t)
      =
      L_0 + A T_t^{-\alpha}
      - B\sum_{i\le t}\Delta_i^+
      \left[1-\left(1+C\,\eta_i^{-\gamma}(T_t-T_i+\eta_i)\right)^{-\beta}\right].
  \]
  Here $\Delta_i^+=\max(\eta_{i-1}-\eta_i,0)$ extracts positive LR decay sources.
\end{frame}

\begin{frame}{MPL Implementation Adaptation}
  The original MPL repository could not be used directly on the course CSV form.
  \begin{itemize}
    \item Its loader reconstructs learning-rate schedules from hard-coded paper filenames.
    \item The cleaned implementation uses the course \code{lr} column directly.
    \item Original-compatible adapter checks showed the formula/evaluator agrees numerically with the upstream evaluator when given the same parameters.
  \end{itemize}
  This separates a data-adapter problem from the mathematical model.
\end{frame}

\begin{frame}{Baseline 3: Faithful FSL}
  Functional Scaling Laws use intrinsic time:
  \[
    T(k)=\sum_{i\le k}\eta_i .
  \]
  The practical LLM ansatz from Appendix B.2 is
  \[
    L_{\Theta}(k)=L_0+\frac{c_1}{T(k)^s}-\mathrm{LRD}(k),
  \]
  where $\mathrm{LRD}$ is a schedule-response convolution over LR drops.
\end{frame}

\begin{frame}{FSL Practical Response}
  In fixed model-size course data, the implemented response is
  \[
  \mathrm{LRD}(k)=
  c_2 \sum_{i\le k}(\eta_{i-1}-\eta_i)^+
  \left(c_3+\frac{1}{T(i)^s}\right)
  \left(1-\frac{1}{(1+c_4(T(k)-T(i)))^\gamma}\right).
  \]
  We do not invent missing kernel-regression quantities such as spectra, source coefficients, or noise variance.
\end{frame}

\begin{frame}{FSL Reproduction Policy}
  Faithful FSL is freshly fit from the audited reproduction formula.
  \begin{itemize}
    \item The code contains the FSL fitter and runs it by default.
    \item The final table uses a fixed-seed 10-restart fit to control local optimizer drift.
    \item The fitted model uses cosine only; WSD loss is used only for evaluation.
  \end{itemize}
  This keeps the result generated from code while preserving the formula implementation.
\end{frame}

\begin{frame}{Unified Baseline Outputs}
  Every baseline writes the same result schema:
  \begin{itemize}
    \item \code{predictions.csv}: method, schedule, role, step, lr, raw/EMA loss, prediction, errors.
    \item \code{metrics.csv}: protocol, method, schedule, target, region, $R^2$, MAE, RMSE, FinalE, WorstE.
    \item \code{fitted\_params.json}: parameters, fit target, source stride, and selection-signal notes.
  \end{itemize}
\end{frame}

\section{Method Development}

\begin{frame}{Development Goal}
  We explored methods that keep the same input/output protocol but change the schedule-response bias.
  \begin{itemize}
    \item Can MPL-style power response improve noisy raw-loss behavior?
    \item Can FSL-style intrinsic-time kernels improve transfer?
    \item Can multiple memory scales beat single-$\lambda$ MTL?
    \item Can a tuned MTL optimizer/hyperparameter setting recover missed performance?
  \end{itemize}
\end{frame}

\begin{frame}{FSL-MPL+ Small}
  FSL-MPL+ small replaces MPL's response with a simpler intrinsic-time power response:
  \[
    R_{\mathrm{small}}(t,i)
    =
    1-\left(1+\kappa \eta_i^{-\gamma}(T_t-T_i)\right)^{-q}.
  \]
  Prediction form:
  \[
    \widehat L(t)=L_0+A T_t^{-\alpha}
    -B\sum_i \Delta_i^+ R_{\mathrm{small}}(t,i).
  \]
  The intent is to reduce purely empirical coupling while retaining LR-drop response.
\end{frame}

\begin{frame}{FSL-MPL+ Source}
  The source variant adds a source-amplitude term:
  \[
    \widehat L(t)=L_0+A T_t^{-\alpha}
    -B\sum_i \Delta_i^+
    \left(1+c_s T_i^{-\alpha}\right)R_{\mathrm{small}}(t,i).
  \]
  Motivation:
  \begin{itemize}
    \item LR drops at different intrinsic times may have different effective strength.
    \item The source term borrows FSL's view that earlier/later sources have different decay behavior.
  \end{itemize}
\end{frame}

\begin{frame}{Kernelized MTL}
  KMTL replaces one exponential memory scale by a nonnegative mixture:
  \[
    S_K(t)=\sum_{j=1}^m w_j S_{\lambda_j}(t),
    \qquad
    w_j\ge 0,\quad \sum_j w_j=1.
  \]
  Tested fixed kernels:
  \begin{itemize}
    \item $m=2$: $\{0.99,0.999\}$.
    \item $m=3$: $\{0.99,0.999,0.9999\}$.
  \end{itemize}
  The hope was to approximate heavy-tail memory with a small mixture.
\end{frame}

\begin{frame}{NCPL-Lite Residual}
  NCPL-lite is a small residual diagnostic, not a full neural ansatz.
  \[
    \log L_{\mathrm{true}}(t) - \log L_{\mathrm{base}}(t)
    \approx
    r_\theta(\phi(t)).
  \]
  Implementation:
  \begin{itemize}
    \item Fit FSL-MPL+ source first.
    \item Fit a ridge residual on cosine features only.
    \item Clip residual corrections to prevent runaway extrapolation.
  \end{itemize}
\end{frame}

\begin{frame}{Tuned MTL}
  The final tuned method keeps the MTL formula unchanged.
  \begin{itemize}
    \item Wider $\lambda$ grid includes $\lambda=0.9997$.
    \item Tail-weighted cosine fitting uses tail weight 3.0.
    \item Same four fitted parameters: $(L_0,A,C,\alpha)$.
  \end{itemize}
  Tuned parameters:
  \[
    L_0=2.7284,\quad A=774.10,\quad C=4.93\times 10^{-5},\quad
    \alpha=0.9343,\quad \lambda=0.9997.
  \]
\end{frame}

\begin{frame}{Tuned MTL Disclosure}
  Tuned MTL is numerically strongest, but its selection is not a clean WSD holdout.
  \begin{itemize}
    \item The autoresearch loop evaluated WSD RMSE repeatedly.
    \item Attempts were kept/discarded based on WSD improvement.
    \item Therefore WSD acted as an adaptive development signal.
  \end{itemize}
  We present it as an important development result: the original MTL family was under-optimized.
\end{frame}

\begin{frame}{What We Did Not Ship}
  The final repository does not include the raw 296-attempt autoresearch loop.
  \begin{itemize}
    \item Raw journals are useful during discovery but noisy for a final project repository.
    \item The final repo ships tuned-MTL search-setting metadata, not tuned fitted weights.
    \item Baselines, variants, FSL, NCPL-lite, and tuned MTL are freshly fit with fixed seeds.
  \end{itemize}
  This keeps the repo presentable while preserving the result logic.
\end{frame}

\section{Failure Exploration}

\begin{frame}{Failure Pattern 1: More Restarts Alone}
  The autoresearch logs show that simply increasing MTL restarts does not explain the final gain.
  \begin{itemize}
    \item 50, 100, and 200 restarts with the default grid stay near the original MTL basin.
    \item The key was the wider $\lambda$ grid plus a stronger tail-weighted fit.
  \end{itemize}
  Lesson: optimizer effort helps only if the search space contains the useful basin.
\end{frame}

\begin{frame}{Failure Pattern 2: Overexpressive Schedule Response}
  MPL and FSL-MPL+ introduce extra nonlinear degrees of freedom.
  \begin{itemize}
    \item These can reduce some average errors.
    \item But they often produce larger endpoint or worst-case errors.
    \item With one fit curve, parameter identifiability is the bottleneck.
  \end{itemize}
\end{frame}

\begin{frame}{Failure Pattern 3: Kernelized MTL Instability}
  KMTL tries to approximate multi-timescale memory.
  \begin{itemize}
    \item $m=2$ can collapse into poor local minima.
    \item $m=3$ has better final error but worse full RMSE than MTL.
    \item Mixture weights are hard to identify from one cosine curve.
  \end{itemize}
\end{frame}

\begin{frame}{Failure Pattern 4: Residual Correction Overreach}
  NCPL-lite has the lowest non-tuned MAE but poor final/worst errors.
  \begin{itemize}
    \item Residual features learn cosine residual structure.
    \item The learned residual does not extrapolate cleanly to WSD tail/final behavior.
    \item Clipping prevents catastrophic blow-up but does not solve calibration.
  \end{itemize}
\end{frame}

\begin{frame}{Failure Pattern 5: Tail Weight Sensitivity}
  Tail weighting matters.
  \begin{itemize}
    \item Low tail weight can overfit the broad stable region.
    \item High tail weight can improve WSD transfer for MTL.
    \item Too much tail emphasis can hurt final or tail-specific objectives.
  \end{itemize}
  Tuned MTL uses tail weight 3.0 because the WSD-adaptive search found it effective for full RMSE.
\end{frame}

\begin{frame}{What The Failures Teach}
  The negative results are useful.
  \begin{itemize}
    \item Schedule-transfer prediction is strongly constrained by data scarcity.
    \item The best formula is not necessarily the most expressive formula.
    \item Hyperparameter grids can dominate the apparent performance of a simple model.
    \item A transparent leakage policy is necessary when method search touches WSD.
  \end{itemize}
\end{frame}

\begin{frame}{Limitations}
  \begin{itemize}
    \item Only three course curves are available.
    \item Main protocol has one fit schedule, so complex formulas are underidentified.
    \item Tuned MTL uses WSD-adaptive model selection.
    \item We do not claim a new theoretical law; the contribution is reproducible comparison and a transparent tuning finding.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
  \begin{itemize}
    \item MTL is the best clean non-WSD-adaptive baseline under the strict protocol.
    \item MPL/FSL/FSL-MPL+/KMTL/NCPL-lite expose useful modeling directions but are not uniformly better on this tiny data regime.
    \item Tuned MTL achieves the strongest WSD full-region RMSE by finding a better $\lambda$ basin and tail weight.
    \item The tuned result should be read as WSD-adaptive development, with 8-1-1 as auxiliary support.
  \end{itemize}
\end{frame}
